This document is used to introduce how to install nnstreamer and NCC elements.

# Step 1: Prerequisites

$ sudo apt-get install meson ninja-build gcc g++ libglib2.0-dev libgstreamer1.0-dev libgstreamer-plugins-base1.0-dev

# Step 2: Installing NNStreamer (opens new window)

#clone
$ git clone https://github.com/nnstreamer/nnstreamer.git

#patch
$ cd nnstreamer
$ cp ~/openncc_frame/nnstreamer/0001-Filter-Decoder-Add-option-for-ncc-and-fix-bug-of-box.patch ./
$ git apply 0001-Filter-Decoder-Add-option-for-ncc-and-fix-bug-of-box.patch

#build and install
$ meson build
$ sudo ninja -C build install

# Step 3: Installing NCC elements into NNStreamer

#build and install
$ cd openncc_frame/nnstreamer
$ meson build
$ sudo ninja -C build install
